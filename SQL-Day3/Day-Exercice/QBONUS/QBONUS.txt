SELECT e.emp_no, e.first_name, e.last_name, e.birth_date, e.gender, e.hire_date, s.salary, s.from_date, s.to_date, t.title, t.from_date, t.to_date, de.dept_no, de.from_date, de.to_date, d.dept_name
FROM employees e 
JOIN salaries s ON e.emp_no = s.emp_no 
JOIN titles t ON e.emp_no = t.emp_no
JOIN dept_emp de ON e.emp_no = de.emp_no
JOIN departments d ON de.dept_no = d.dept_no 
GROUP BY emp_no;
